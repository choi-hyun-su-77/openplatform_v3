8. 가장 많이 쓰는 명령

# 프로젝트 디렉토리
cd C:/claude/openplatform_v3

# 상태 확인
./scripts/start.sh status

# 전체 기동 (healthcheck + resources)
./scripts/start.sh full

# 관측성 포함 (Loki/Grafana/Prometheus)
./scripts/start.sh observability

# 운영 모드 (전체 override)
./scripts/start.sh production

# 중지
./scripts/start.sh stop

# 특정 서비스 로그 tail
./scripts/start.sh logs v3-backend-core

# Mongo replica set 초기화 (최초 1회)
./scripts/start.sh init-mongo
